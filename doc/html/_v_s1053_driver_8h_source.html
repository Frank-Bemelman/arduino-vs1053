144  int setSampleRate(int sample_rate){
145  int diff_min = sample_rate; // max value for difference
146  int sample_rate_result = -1;
147  multiplier = -1;
148 
149  // find the combination with the smallet difference
150  for (int j=0;j<7;j++){
151  for (int div=4;div<=126;div++){
152  float mf = multiplier_factors[j];
153  uint16_t sc_mult = sc_multipliers[j];
154  int sample_rate_eff = getSampleRate(div, mf);
155 
156  if (sample_rate_eff < sample_rate){
157  // increasing dividers will make the sample rate just smaller, wo we break
158  break;
159  }
160 
161  int diff = abs(sample_rate_eff - sample_rate);
162  LOG("--> div: %d - mult: %f (%x) -> %d", div, mf, sc_mult, sample_rate_eff);
163  if (diff<diff_min){
164  LOG("==> div: %d - mult: %f (%x) -> %d", div, mf, sc_mult, sample_rate_eff);
165  diff_min = diff;
166  this->divider = div;
167  this->multiplier = sc_mult;
168  this->multiplier_factor = mf;
169  sample_rate_result = sample_rate_eff;
170  }
171  // if we found the correct rate we are done
172  if (diff==0) return sample_rate_result;
173  }
174  }
175  return multiplier==-1? -1: sample_rate_result;
176  }
